/users
    /getAll
        - Returns all users
            PARAMS: None

    /create
        - Creates a new user
            PARAMS: email:string, password:string

    /auth
        - Authenticates a user to the system
            PARAMS: username: string, password: string 
    
    /getUserRecord
        - Loads all the user data from database (User Info, League, and Roster records)
            PARAMS: email: string
